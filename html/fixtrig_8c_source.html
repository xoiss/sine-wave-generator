130 /* Returns the sine given a momentary phase, signed fixed point 0.15-bit version. */
131 uq016_t qsin_uq016(const uq016_t phi) {
132 
134  #define _PHI_RANK (POW2(UQ016_BIT) / 4) /* Number of different phi values in the first quadrant. */
135  #define _KEY_RANK (ARRAY_SIZE(qsin_lut)) /* Number of entries in the phase-to-sine LUT. */
136  #define _COEF_RANK (_PHI_RANK / _KEY_RANK) /* Number of different phi values between LUT entries. */
137  #define _COEF_BIT (LOG2(_COEF_RANK)) /* Width of the linear interpolation coefficient. */
138  #define _COEF_MASK (BIT_MASK(_COEF_BIT)) /* Bit mask for linear interpolation coefficient. */
139  #define _1 (0x0000u) /* Container value for UQ0.16 value 1.0
140  * represented as 0.0 modulo 1.0. */
141 
143  ui8_t key0; /* Left side key into the phase-to-sine LUT. */
144  uq016_t coef; /* Linear interpolation coefficient. */
145 
146  static_assert(POW2(sizeof(key0) * 8) == _KEY_RANK);
147 
148  key0 = phi >> _COEF_BIT;
149  coef = (phi & _COEF_MASK) << (UQ016_BIT - _COEF_BIT);
150 
151  if (coef == 0) {
152  return qsin_lut[key0];
153 
154  } else {
155  ui8_t key1; /* Right side key into the phase-to-sine LUT. */
156  uq016_t val0, val1; /* Left and right side values taken from the LUT with linear weight. */
157  key1 = key0 + 1;
158  val1 = key1 == 0 ? coef : qmul_uq016(qsin_lut[key1], coef);
159  val0 = qmul_uq016(qsin_lut[key0], _1 - coef);
160  return val0 + val1;
161  }
162 
163  #undef _PHI_RANK
164  #undef _KEY_RANK
165  #undef _COEF_RANK
166  #undef _COEF_BIT
167  #undef _COEF_MASK
168  #undef _1
169 }

172 /* Returns the modulated sine given a momentary phase and momentary attenuation factor, signed fixed point 0.15-bit
173  * version. */
174 sq015_t msin_sq015(const uq016_t phi, const uq016_t att) {
175 
177  #define _PI2 (0x4000u) /* Container value for UQ0.16 value 0.25 which stays for pi/2 radian. */
178  #define _PI (0x8000u) /* Container value for UQ0.16 value 0.5 which stays for pi radian. */
179  #define _3PI2 (0xC000u) /* Container value for UQ0.16 value 0.75 which stays for 3*pi/2 radian. */
180  #define _1P (0x7FFF) /* Container value for SQ0.15 value +1.0-1/2^15. */
181  #define _1N (0x8000) /* Container value for SQ0.15 value -1.0. */
182  #define _1 (0x0000u) /* Container value for UQ0.16 value 1.0 represented as 0.0 modulo 1.0. */
183 
185  if (phi == _PI2) {
186  return att == 0 ? _1P : +sq015_from_uq016(_1 - att);
187 
188  } else if (phi == _3PI2) {
189  return att == 0 ? _1N : -sq015_from_uq016(_1 - att);
190 
191  } else {
192  uq016_t phi1 = phi; /* Value of phi brought into the first quadrant - i.e., the range [0; pi/2) radian. */
193  bool_t neg = 0; /* Equals to 1 if sin(phi) < 0; equals to 0 if sin(phi) >= 0. */
194  uq016_t usin; /* Unsigned 0.16-bit absolute value of sin(phi). */
195  bool_t lsb; /* Value of LSB of 0.16-bit value returned by sin(phi) before rounding to 0.15-bit. */
196  sq015_t ssin; /* Signed 0.15-bit absolute value of sin(phi). */
197 
198  if (phi >= _PI) {
199  phi1 -= _PI;
200  neg = 1;
201  }
202  if (phi1 > _PI2) {
203  phi1 = _PI - phi1;
204  }
205 
206  usin = qsin_uq016(phi1);
207 
208  if (att > 0) {
209  usin = qmul_uq016(usin, _1 - att);
210  }
211 
212  lsb = usin & 1;
213  ssin = sq015_from_uq016(usin);
214  if (lsb && ssin < 0x7FFF) {
215  ++ssin;
216  }
217 
218  return neg ? -ssin : +ssin;
219  }

222  #undef _PI
223  #undef _3PI2
224  #undef _1P
225  #undef _1N
226  #undef _1
227 }
